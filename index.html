import java.io.*;
import java.util.*;

public class Contactos {
    private static Scanner scanner = new Scanner(System.in);
    private static List<String[]> contactos = new ArrayList<>();

    public static void guardarContactos(List<String[]> contactos, String archivo) throws IOException {
        try (BufferedWriter bw = new BufferedWriter(new FileWriter(archivo))) {
            for (String[] contacto : contactos) {
                bw.write(contacto[0] + "," + contacto[1]);
                bw.newLine();
            }
        }
    }

    public static void mostrarContactos(String archivo) throws IOException {
        System.out.println("Lista de contactos:");
        try (BufferedReader br = new BufferedReader(new FileReader(archivo))) {
            String linea;
            while ((linea = br.readLine()) != null) {
                String[] partes = linea.split(",");
                System.out.println("Nombre: " + partes[0] + ", Teléfono: " + partes[1]);
            }
        }
    }

    public static void agregarContacto() {
        System.out.print("Ingrese el nombre: ");
        String nombre = scanner.nextLine();
        
        System.out.print("Ingrese el teléfono: ");
        String telefono = scanner.nextLine();
        
        String[] nuevoContacto = {nombre, telefono};
        contactos.add(nuevoContacto);
        
        System.out.println(" Contacto agregado correctamente!");
    }

    public static void mostrarMenu() {
        System.out.println("\n=== MENU DE CONTACTOS ===");
        System.out.println("1. Agregar contacto");
        System.out.println("2. Mostrar contactos");
        System.out.println("3. Guardar y salir");
        System.out.print("Seleccione una opción: ");
        System.out.println("\n=========================");
    }

    public static void main(String[] args) {
        String archivo = "contactos.txt";
        int opcion;
        
        do {
            mostrarMenu();
            opcion = scanner.nextInt();
            scanner.nextLine(); 
            
            try {
                switch (opcion) {
                    case 1:
                        agregarContacto();
                        break;
                    case 2:
                        mostrarContactos(archivo);
                        break;
                    case 3:
                        guardarContactos(contactos, archivo);
                        System.out.println(" Contactos guardados en " + archivo);
                        break;
                    default:
                        System.out.println(" Opcion no valida");
                }
            } catch (IOException e) {
                System.out.println("Error: " + e.getMessage());
            }
        } while (opcion != 3);
        
        scanner.close();
    }
}

Clase File de Java

La clase File en Java representa archivos y directorios en el sistema de archivos. Permite realizar operaciones como crear, eliminar, renombrar y obtener información sobre archivos y directorios.

Métodos Principales de la Clase File

Método Descripción Ejemplo
String getName() Devuelve el nombre del archivo o directorio archivo.getName()
String getPath() Devuelve la ruta relativa del archivo archivo.getPath()
String getAbsolutePath() Devuelve la ruta absoluta del archivo archivo.getAbsolutePath()
boolean exists() Verifica si el archivo o directorio existe archivo.exists()
boolean canWrite() Verifica si la aplicación puede escribir en el archivo archivo.canWrite()
boolean canRead() Verifica si la aplicación puede leer el archivo archivo.canRead()
boolean isFile() Verifica si es un archivo (no un directorio) archivo.isFile()
boolean isDirectory() Verifica si es un directorio archivo.isDirectory()
boolean isAbsolute() Verifica si la ruta es absoluta archivo.isAbsolute()
long lastModified() Devuelve la fecha de última modificación en milisegundos archivo.lastModified()
long length() Devuelve el tamaño del archivo en bytes archivo.length()
boolean mkdir() Crea el directorio especificado archivo.mkdir()
boolean mkdirs() Crea el directorio y todos los directorios padres necesarios archivo.mkdirs()
boolean renameTo(File dest) Renombra el archivo o directorio archivo.renameTo(nuevoArchivo)
boolean delete() Elimina el archivo o directorio archivo.delete()
String[] list() Devuelve un array con los nombres de archivos en el directorio archivo.list()
String[] list(FilenameFilter filter) Devuelve un array con los nombres de archivos que cumplen el filtro archivo.list(filtro)

Ejemplo de Uso de la Clase File

import java.io.File;

public class EjemploFile {
    public static void main(String[] args) {
        // Crear un objeto File
        File archivo = new File("contactos.txt");
        
        // Verificar si el archivo existe
        if (archivo.exists()) {
            System.out.println("El archivo existe");
            System.out.println("Nombre: " + archivo.getName());
            System.out.println("Ruta absoluta: " + archivo.getAbsolutePath());
            System.out.println("Tamaño: " + archivo.length() + " bytes");
            System.out.println("Se puede leer: " + archivo.canRead());
            System.out.println("Se puede escribir: " + archivo.canWrite());
        } else {
            System.out.println("El archivo no existe");
        }
        
        // Crear un directorio
        File directorio = new File("mis_archivos");
        if (directorio.mkdir()) {
            System.out.println("Directorio creado exitosamente");
        }
        
        // Listar archivos en un directorio
        File carpetaActual = new File(".");
        String[] archivos = carpetaActual.list();
        System.out.println("Archivos en el directorio actual:");
        for (String nombre : archivos) {
            System.out.println(nombre);
        }
    }
}
Nota Importante

La clase File representa una ruta abstracta que puede o no existir en el sistema de archivos. Muchos de sus métodos devuelven valores booleanos que indican si la operación fue exitosa. Es importante verificar estos valores para manejar correctamente posibles errores.
