import java.io.*;
import java.util.*;

public class Contactos {
    private static Scanner scanner = new Scanner(System.in);
    private static List<String[]> contactos = new ArrayList<>();

    public static void guardarContactos(List<String[]> contactos, String archivo) throws IOException {
        try (BufferedWriter bw = new BufferedWriter(new FileWriter(archivo))) {
            for (String[] contacto : contactos) {
                bw.write(contacto[0] + "," + contacto[1]);
                bw.newLine();
            }
        }
    }

    public static void mostrarContactos(String archivo) throws IOException {
        System.out.println("Lista de contactos:");
        try (BufferedReader br = new BufferedReader(new FileReader(archivo))) {
            String linea;
            while ((linea = br.readLine()) != null) {
                String[] partes = linea.split(",");
                System.out.println("Nombre: " + partes[0] + ", Tel√©fono: " + partes[1]);
            }
        }
    }

    public static void agregarContacto() {
        System.out.print("Ingrese el nombre: ");
        String nombre = scanner.nextLine();
        
        System.out.print("Ingrese el tel√©fono: ");
        String telefono = scanner.nextLine();
        
        String[] nuevoContacto = {nombre, telefono};
        contactos.add(nuevoContacto);
        
        System.out.println(" Contacto agregado correctamente!");
    }

    public static void mostrarMenu() {
        System.out.println("\n=== MENU DE CONTACTOS ===");
        System.out.println("1. Agregar contacto");
        System.out.println("2. Mostrar contactos");
        System.out.println("3. Guardar y salir");
        System.out.print("Seleccione una opci√≥n: ");
        System.out.println("\n=========================");
    }

    public static void main(String[] args) {
        String archivo = "contactos.txt";
        int opcion;
        
        do {
            mostrarMenu();
            opcion = scanner.nextInt();
            scanner.nextLine(); 
            
            try {
                switch (opcion) {
                    case 1:
                        agregarContacto();
                        break;
                    case 2:
                        mostrarContactos(archivo);
                        break;
                    case 3:
                        guardarContactos(contactos, archivo);
                        System.out.println(" Contactos guardados en " + archivo);
                        break;
                    default:
                        System.out.println(" Opcion no valida");
                }
            } catch (IOException e) {
                System.out.println("Error: " + e.getMessage());
            }
        } while (opcion != 3);
        
        scanner.close();
    }
}
